def update_config(keyword, new_value):
    # Step 1: Read the file and store lines in a list
    with open("server.conf", "r") as file:
        lines = file.readlines()

    # Step 2: Modify the lines
    updated_lines = []
    for line in lines:
        if line.startswith(keyword):
            line = f"{keyword}={new_value}\n"
        updated_lines.append(line)

    # Step 3: Write the updated content back to the file
    with open("server.conf", "w") as file:
        file.writelines(updated_lines)

# Example usage
update_config("port", "9090")
